Rendering List means displaying multiple items from array using jsx

const items = ["Space suit", "Helmet", "Photo of Tam"];
function PackingList() {
  return (
    <ul>
      {items.map(item => (
        <li key={item}>{item}</li>
      ))}
    </ul>
  );
}
